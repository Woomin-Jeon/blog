2020-02-21

```javascript
const solution = (array, commands) => {
  const answer = [];
  let tempt = [];

  commands.forEach((v, i) => {
    tempt = array.slice((commands[i][0] - 1), commands[i][1]).sort((a, b) => a - b);
    answer.push(tempt[commands[i][2] - 1]);
  });
  
  return answer;
};
```
