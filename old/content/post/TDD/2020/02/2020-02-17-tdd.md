## 성공

2020-02-17

```javascript
const solution = (n, m) => {
  return [Gcd(n, m), Lcm(n, m)];
};

const BiggerNum = (n, m) => n > m ? n : m; 
const SmallerNum = (n, m) => n < m ? n : m;

const Lcm = (n, m) => {
  for (let i = BiggerNum(n, m); i <= n * m; i++) {
    if (i % n === 0 && i % m === 0) {
      return i;
    }
  }
} // 최소공배수 = (n * m) / 최대공약수

const Gcd = (n, m) => {
  for (let i = SmallerNum(n, m); i >= 1 ; i--) {
    if (n % i === 0 && m % i === 0) {
      return i;
    }
  }
}
```

const solution = (n, m) => [Gcd(n, m), (n * m) / Gcd(n, m)];

const Gcd = (n, m) => {
  for (let i = n < m ? n : m; i >= 1 ; i--) {

      return i;
    }
  }
}
