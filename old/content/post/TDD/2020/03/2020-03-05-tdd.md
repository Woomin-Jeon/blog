## 실패

2020-03-05

```javascript
const solution =  (n) => {  
  const changeBinary = n.toString(3);
  const arr = changeBinary.split('');

  const changeNumberByRule = arr.map(v =>
    v == 0 ? v = 1
    : v == 1 ? v = 2
      : v == 2 ? v = 4 : v
    );
  
  return parseInt(changeNumberByRule.join(''));
};
```

## 성공

2020-03-09

```javascript
const solution = (n) => {
  const answer = [];

  while(n > 0) {
    if (n % 3 === 0) {
      answer.unshift(4);
      n = Math.floor(n / 3) - 1;
    } else {
      answer.unshift(n % 3);
      n = Math.floor(n / 3);
    }
  }

  return answer.join('');
};
```
