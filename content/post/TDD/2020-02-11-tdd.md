const solution = (heights) => {
  const answer = [];

  for (let i = 0; i < heights.length; i++) {
    answer.push(biggerThan(heights, i));
  }

  return answer;
};

const biggerThan = (heights, n) => {
  for (let i = n - 1; i >= 0; i--) {
    if (heights[n] < heights[i]) {
      return i + 1;
    }
    return 0; // 왜... return을... 여기에다가...
  }
};
